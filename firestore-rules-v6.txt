rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{uid} { allow read, write: if request.auth != null; }
    match /progress/{uid} { allow read, write: if request.auth != null && request.auth.uid == uid; }
    match /activity/{uid}/items/{docId} { allow read, write: if request.auth != null && request.auth.uid == uid; }
  }
}
